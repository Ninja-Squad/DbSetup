025    package com.ninja_squad.dbsetup;
026    
027    /**
028     * A runtime exception indicating that a DbSetup failed.
029     * @author JB Nizet
030     */
031    public class DbSetupRuntimeException extends RuntimeException {
032    
033        public DbSetupRuntimeException() {
034            super();
035        }
036    
037        public DbSetupRuntimeException(String message, Throwable cause) {
038            super(message, cause);
039        }
040    
041        public DbSetupRuntimeException(String message) {
042            super(message);
043        }
044    
045        public DbSetupRuntimeException(Throwable cause) {
046            super(cause);
047        }
048    }
